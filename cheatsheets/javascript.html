/**
 *   Mathématique
 */

Expr + Expr     // addition
Expr - Expr     // soustraction
Expr * Expr     // multiplication
Expr / Expr     // division
Expr % Expr     // reste de la division

/**
 *   Comparaison
 */

Expr == Expr        // égalité
Expr != Expr        // différent
Expr < Expr         // inférieur
Expr <= Expr        // inférieur ou égal
Expr > Expr         // supérieur
Expr >= Expr        // supérieur ou égal

/**
 *   Logique
 */

Expr && Expr        // et
Expr || Expr        // ou
! Expr              // non

/**
 *   If ...
 */

if ( CONDITION ) {
    ...                 // done if cond is true
}


/**
 *   If ... Else ...
 */

if ( CONDITION ) {
    ...                 // done if cond is true
} else {
    ...                 // done if cond is false
}

/**
 *   Switch Statement
 */

switch ( EXPR ) {
    case VALEUR1:               // if EXPR == VALEUR1
        ...
        break;
    case VALEUR2 || VALEUR3:    // if EXPR == VALEUR2
        ...                     // or EXPR == VALEUR3
        break;
    ...
    default:                    // else...
        ...
}




/**
 *   While Loop
 */

while ( CONDITION ) {
    ...
}
// exécute le code tant que la condition est vraie
// peut ne pas rentrer dans la boucle

/**
 *   Do While Loop
 */

do {
    ...
} while ( condition )
// exécute le code tant que la condition est vraie
// passe au moins une fois dans la boucle

/**
 *   For Loop
 */

for ( INIT ; COND ; INCR ) {
    ...
}
// commence par exécuter INIT
// puis exécute le code un certain nombre de fois
// en exécutant INCR à chaque tour

for ( var NOM in OBJET ) {
    ...
}
// exécute le code pour chaque attribut de OBJET
// NOM prendra la valeur de chaque attribut successivement

/**
 *   Break
 */

BOUCLE {
    ...
    break;          // arrète l'exécution de la boucle
    ...
}

/**
 *   Continue
 */

BOUCLE {
    ...
    continue;       // arrète l'exécution de ce tour de boucle
    ...
}

    ...
